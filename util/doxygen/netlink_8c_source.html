128 char *flags2str(uint32_t bits, const struct flags *flags, size_t size)
129 {
130  char *str = malloc(size);
131  str[0] = '<';
132  str[1] = '\0';
133 
134  /* Iterate until the end of bits or thh end of flags array */
135  for (int i = 0; i < sizeof(uint32_t) * 8 && flags[i].name != 0; ++i) {
136  // extract the i-th bit
137  int b = ((bits >> i) & 1);
138  // b will be 1 if i-th bit is set, 0 otherwise
139 
140  if (b == 1) {
141  strcat(str, flags[i].name);
142  strcat(str, ",");
143  }
144  }
145  int len = strlen(str);
146 
147  if (len <= 1) {
148  strcat(str, ">");
149  } else {
150  str[len-1] = '>'; /* replace ',' with '>' */
151  }
152 
153  return str;
154 }

221 int ce_gw_add(char *src_name, char *dst_name, uint8_t type, uint32_t flags)
222 {
223  int err = 0;
224  struct nl_msg *msg;
225 
226  /* create */
227  msg = nlmsg_alloc();
228  if(msg == NULL) {
229  fprintf(stderr,"add: Message allocation failed.\n");
230  return -ENOMEM;
231  }
232 
233  void *user_hdr;
234  user_hdr = genlmsg_put(msg, NL_AUTO_PORT, NL_AUTO_SEQ,
235  genl_family_get_id(genl_fam), USER_HDR_SIZE,
236  NO_FLAG, CE_GW_C_ADD, IFACE_VERSION);
237  if (user_hdr == NULL)
238  fprintf(stderr, "add: Message Haeder creation failed\n");
239 
240  if (src_name != NULL) { /* indicates that it is add route command */
241  NLA_PUT_STRING(msg, CE_GW_A_SRC, src_name);
242  NLA_PUT_STRING(msg, CE_GW_A_DST, dst_name);
243  } /* else it is add dev command */
244 
245  /* Attributes needed by both, add route and add dev */
246  NLA_PUT_U8(msg, CE_GW_A_TYPE, type);
247  NLA_PUT_U32(msg, CE_GW_A_FLAGS, flags);
248 
249  /* vaildate */
250  struct nlmsghdr *msghdr = nlmsg_hdr(msg);
251  err = genlmsg_validate(msghdr, USER_HDR_SIZE,
252  CE_GW_A_MAX, ce_gw_genl_policy);
253  if (err != 0) {
254  fprintf(stderr, "add: Validation of Message Failed: %i\n", err);
255  return -1;
256  }
257 
258  /* send */
259  nl_send_auto(nl_sk, msg);
260 
261  err = nl_wait_for_ack(nl_sk);
262  if (err != 0) {
263  fprintf(stderr,
264  "add: ACK is missing or Error returned. "
265  "Operation might fail: %i\n", err);
266  }
267 
268  nlmsg_free(msg);
269 
270  return 0;

567  fprintf (stderr, "Error Callback modification failed: %d\n",
568  err);
569  }
570 
571  /* create generic netlink family */
572  genl_fam = genl_family_alloc();
573  if (genl_fam == NULL) {
574  fprintf (stderr, "Family allocation failed.\n");
575  return -1;
576  }
577

579  genl_family_set_version(genl_fam, GE_FAMILY_VERSION);
580  genl_family_set_maxattr(genl_fam, CE_GW_A_MAX);

584  (err = genl_ctrl_resolve(nl_sk, GE_FAMILY_NAME)));
