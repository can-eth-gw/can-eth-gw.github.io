237  struct net_device *dev;
238 
239  dev = alloc_netdev(sizeof(struct ce_gw_job_info),
240  "cegw%d", ether_setup);
241  if (dev == NULL) {
242  pr_err("ce_gw_dev: Error allocation etherdev.");
243  goto ce_gw_dev_create_error;
244  }
245 
246  /* initialize private field */
247  struct ce_gw_job_info *priv = netdev_priv(dev);
248  memset(priv, 0, sizeof(struct ce_gw_job_info));
249  priv->job_src.first = NULL;
250  priv->job_dst.first = NULL;
251 
252  /* create list entry and add */

254  dl = kmem_cache_alloc(ce_gw_dev_cache, GFP_KERNEL);
255  if (dl == NULL) {
256  pr_err("ce_gw_dev: cache alloc failed");
257  goto ce_gw_dev_create_error_cache;
258  }
259 
260  dl->dev = dev;
261 
262  hlist_add_head_rcu(&dl->list_alloc, &ce_gw_dev_allocated);
263 
264  return dev;
265 
266 ce_gw_dev_create_error_cache:
267  kmem_cache_free(ce_gw_dev_cache, dl);
268 
269 ce_gw_dev_create_error:
270  free_netdev(dev);
271  return NULL;
272 }

303 void ce_gw_dev_setup(struct net_device *dev, enum ce_gw_type type,
304  __u32 flags) {
305  dev->netdev_ops = &ce_gw_ops;
306 
307  /* Set sensible MTU */
308  switch (type) {
309  case CE_GW_TYPE_NONE:
310  /* Do Nothing (default Ethernet MTU will be set) */
311  break;
312  case CE_GW_TYPE_ETH:
313  if ((flags & CE_GW_F_CAN_FD) == CE_GW_F_CAN_FD) {
314  dev->mtu = CANFD_MAX_DLEN;
315  } else {
316  dev->mtu = CAN_MAX_DLEN;
317  }
318  break;
319  case CE_GW_TYPE_NET:
320  if ((flags & CE_GW_F_CAN_FD) == CE_GW_F_CAN_FD) {
321  dev->mtu = sizeof(struct canfd_frame);
322  } else {
323  dev->mtu = sizeof(struct can_frame);
324  }
325  break;
326  case CE_GW_TYPE_TCP:
327  /* TODO nothing yet */
328  break;
329  case CE_GW_TYPE_UDP:
330  /* TODO nothing yet */
331  break;
332  default:
333  pr_err("ce_gw_dev: Type not defined.");
334  }
335 }
