600 struct sk_buff *ce_gw_can_to_eth(unsigned char *dest, unsigned char *source,
601  __be16 type, struct sk_buff *can_buffer, struct
602  net_device *dev) {
603  struct can_frame *can_frame_skb = (struct can_frame *)
604  can_buffer->data;
605  struct ethhdr *ethhdr;
606  u8 canlen = can_dlc2len(can_frame_skb->can_dlc);
607  struct sk_buff *eth_skb = dev_alloc_skb(sizeof(struct ethhdr) + sizeof
608  (struct can_frame) + 64);
609  if (eth_skb == NULL) {
610  printk (KERN_ERR "ce_gw_main.c: kmalloc failed in function"
611  "ce_gw_can_to_eth \n");
612  return NULL;
613  }
614  eth_skb->dev = dev;
615 
616  /*updates data pointer to tail */
617  skb_reserve(eth_skb, sizeof(struct ethhdr) + sizeof(struct can_frame) +
618  64);
619  /*updates data pointer to start of network header*/
620  eth_skb->data = skb_push(eth_skb, (const int) canlen);
621 
622  /*copys data from can sk buffer into ethernet sk buffer */
623  memcpy(eth_skb->data, can_buffer->data + sizeof(struct can_frame) -
624  - canlen, canlen);
625  /*update network_header */
626  skb_set_network_header(eth_skb, 0);
627  /*updates transport header */
628  skb_set_transport_header(eth_skb, (const int) canlen);
629 
630  /*updates data pointer to mac header */
631  eth_skb->data = skb_push(eth_skb, (const int) sizeof(struct ethhdr));
632  /*updates mac_header*/
633  skb_set_mac_header(eth_skb, 0);
634  /*fills ethhdr with data */
635  ethhdr = eth_hdr(eth_skb);
636  memcpy(ethhdr->h_dest, dest, ETH_ALEN);
637  memcpy(ethhdr->h_source, source, ETH_ALEN);

925 
926  switch (gwj->type) {
927 
928  case CE_GW_TYPE_ETH:
929  pr_info("Translation ETH not implemented yet.");
930  break;
931 
932  case CE_GW_TYPE_NET:
933  can_skb = ce_gw_net2can_alloc(eth_skb, gwj->dst.dev);
934  break;
935 
936  case CE_GW_TYPE_TCP:
937  pr_info("Translation IP TCP not implemented yet.");
938  break;
939 
940  case CE_GW_TYPE_UDP:
941  pr_info("Translation IP UDP not implemented yet.");
942  break;
943 
944  default:
945  printk(KERN_ERR "ce_gw: Translation type of ce_gw_job not "
946  "implemented. BUG: Some module inserted an invalid "
947  "type. Use enum ce_gw_type instead.");
948  goto drop_frame;
949  break;
950  }
951 
952  /* Memory allocation or translation ETH -> CAN failed */
953  if (!can_skb)
954  goto drop_frame;
955 
956  struct can_frame *cf;
957  cf = (struct can_frame *)can_skb->data;
958  pr_debug("Incoming msg from eth dev (gwj %i): "
959  "can_id %x, len %i, can_msg(1) %x\n",
960  gwj->id, cf->can_id, cf->can_dlc, cf->data[0]);
961 
962  /* send to CAN netdevice (with echo flag for loopback devices) */
963  if (can_send(can_skb, 0x01))
964  goto drop_frame;
965  else
966  gwj->handled_frames++;
967 
968  return; /* Receive + process + send to CAN successful */
969 
970 drop_frame:
971  gwj->dropped_frames++;
972  dev_kfree_skb(can_skb);
973  return;
974 }

1025  gwj->dst.dev = dev_get_by_index(&init_net, dst_ifindex);

1037  * Depending on routing direction: register at source device
1038  */
1039  if (gwj->src.dev->type == ARPHRD_CAN &&
1040  ce_gw_is_registered_dev(gwj->dst.dev) == 0) {
1041  /* && gwj->dst.dev->type == ARPHRD_ETHER) {*/
1042  /* CAN source --> ETH destination (cegw virtual dev) */
1043  err = ce_gw_register_can_src(gwj);
1044  } else if (ce_gw_is_registered_dev(gwj->src.dev) == 0 &&
1045  gwj->dst.dev->type == ARPHRD_CAN) {
1046  /* ETH source (cegw virtual dev) --> CAN destination */
1047  err = ce_gw_register_eth_src(gwj);

1049  /* Undefined routing setup */
1050  goto clean_exit;
1051  }
1052 
1053  if (!err)

1061  dev_put(gwj->src.dev);
1062  if (gwj->dst.dev)
