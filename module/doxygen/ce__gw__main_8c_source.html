532 struct sk_buff *ce_gw_can_to_eth(unsigned char *dest, unsigned char *source,
533  __be16 type, struct sk_buff *can_buffer, struct
534  net_device *dev) {
535  struct can_frame *can_frame_skb = (struct can_frame *)
536  can_buffer->data;
537  struct ethhdr *ethhdr;
538  u8 canlen = can_dlc2len(can_frame_skb->can_dlc);
539  struct sk_buff *eth_skb = dev_alloc_skb(sizeof(struct ethhdr) + sizeof
540  (struct can_frame) + 64);
541  if (eth_skb == NULL) {
542  printk (KERN_ERR "ce_gw_main.c: kmalloc failed in function"
543  "ce_gw_can_to_eth \n");
544  return NULL;
545  }
546  eth_skb->dev = dev;
547 
548  /*updates data pointer to tail */
549  skb_reserve(eth_skb, sizeof(struct ethhdr) + sizeof(struct can_frame) +
550  64);
551  /*updates data pointer to start of network header*/
552  eth_skb->data = skb_push(eth_skb, (const int) canlen);
553 
554  /*copys data from can sk buffer into ethernet sk buffer */
555  memcpy(eth_skb->data, can_buffer->data + sizeof(struct can_frame) -
556  - canlen, canlen);
557  /*update network_header */
558  skb_set_network_header(eth_skb, 0);
559  /*updates transport header */
560  skb_set_transport_header(eth_skb, (const int) canlen);
561 
562  /*updates data pointer to mac header */
563  eth_skb->data = skb_push(eth_skb, (const int) sizeof(struct ethhdr));
564  /*updates mac_header*/
565  skb_set_mac_header(eth_skb, 0);
566  /*fills ethhdr with data */
567  ethhdr = eth_hdr(eth_skb);
568  memcpy(ethhdr->h_dest, dest, ETH_ALEN);
569  memcpy(ethhdr->h_source, source, ETH_ALEN);
570  ethhdr->h_proto = type;
571 
572  return eth_skb;
573 }

763  cf = (struct can_frame *)can_skb->data;

768  struct sk_buff *eth_skb = NULL;

772

777  break;

793  break;

828 void ce_gw_eth_rcv(struct sk_buff *eth_skb, void *data)
829 {
830  struct ce_gw_job *gwj = (struct ce_gw_job *)data;
831 
832  /* Create Can skb and convert incoming Eth sk buffer depending on
833  * type (ce_gw_type in gwj)
834  */
835  struct sk_buff *can_skb = NULL;
836 
837  switch (gwj->type) {
838 
839  case CE_GW_TYPE_ETH:
840  pr_info("Translation ETH not implemented yet.");
841  break;
842 
843  case CE_GW_TYPE_NET:
844  can_skb = ce_gw_net2can_alloc(eth_skb, gwj->dst.dev);
845  break;
846 
847  case CE_GW_TYPE_TCP:
848  pr_info("Translation IP TCP not implemented yet.");
849  break;
850 
851  case CE_GW_TYPE_UDP:
852  pr_info("Translation IP UDP not implemented yet.");
853  break;
854 
855  default:
856  printk(KERN_ERR "ce_gw: Translation type of ce_gw_job not "
857  "implemented. BUG: Some module inserted an invalid "
858  "type. Use enum ce_gw_type instead.");
859  goto drop_frame;
860  break;
861  }
862 
863  /* Memory allocation or translation ETH -> CAN failed */
864  if (!can_skb)
865  goto drop_frame;
866 
867  struct can_frame *cf;
868  cf = (struct can_frame *)can_skb->data;
869  pr_debug("Incoming msg from eth dev (gwj %i): "
870  "can_id %x, len %i, can_msg(1) %x\n",
871  gwj->id, cf->can_id, cf->can_dlc, cf->data[0]);
872 
873  /* send to CAN netdevice (with echo flag for loopback devices) */
874  if (can_send(can_skb, 0x01))
875  goto drop_frame;
876  else
877  gwj->handled_frames++;
878 
879  return; /* Receive + process + send to CAN successful */
880 
881 drop_frame:
882  gwj->dropped_frames++;
883  dev_kfree_skb(can_skb);
884  return;
885 }

915 int ce_gw_create_route(int src_ifindex, int dst_ifindex,
916  enum ce_gw_type rt_type, u32 flags)
917 {
918  int err = 0;
919 
920  struct ce_gw_job *gwj;
921  gwj = kmem_cache_alloc(ce_gw_job_cache, GFP_KERNEL);
922 
923  gwj->id = job_count++;
924  gwj->handled_frames = 0;
925  gwj->dropped_frames = 0;
926 
927  err = -ENODEV;
928  gwj->src.dev = dev_get_by_index(&init_net, src_ifindex);
929  gwj->dst.dev = dev_get_by_index(&init_net, dst_ifindex);
930  if (!gwj->src.dev || !gwj->dst.dev)
931  goto clean_exit;
932 
933  gwj->type = rt_type;
934  gwj->flags = flags;
935 
936  /* TEST: pre-filled values */
937  gwj->can_rcv_filter.can_id = 0x42A;
938  gwj->can_rcv_filter.can_mask = 0; /* allow all frames */
939 
940  /*
941  * Depending on routing direction: register at source device
942  */
943  if (gwj->src.dev->type == ARPHRD_CAN &&
944  ce_gw_is_registered_dev(gwj->dst.dev) == 0) {
945  /* && gwj->dst.dev->type == ARPHRD_ETHER) {*/
946  /* CAN source --> ETH destination (cegw virtual dev) */
947  err = ce_gw_register_can_src(gwj);
948  } else if (ce_gw_is_registered_dev(gwj->src.dev) == 0 &&
949  gwj->dst.dev->type == ARPHRD_CAN) {
950  /* ETH source (cegw virtual dev) --> CAN destination */
951  err = ce_gw_register_eth_src(gwj);
952  } else {
953  /* Undefined routing setup */
954  goto clean_exit;
955  }
956 
957  if (!err)
958  hlist_add_head_rcu(&gwj->list, &ce_gw_job_list);
959 
960 clean_exit:
961  if (err) {
962  printk(KERN_ERR "ce_gw: Src or dst device not found or "
963  "not compatible (CAN<->CEGW ETH), exit.\n");
964  if (gwj->src.dev)
965  dev_put(gwj->src.dev);
966  if (gwj->dst.dev)
967  dev_put(gwj->dst.dev);
968  kmem_cache_free(ce_gw_job_cache, gwj);
969  }
970 
971  return err;
972 }
